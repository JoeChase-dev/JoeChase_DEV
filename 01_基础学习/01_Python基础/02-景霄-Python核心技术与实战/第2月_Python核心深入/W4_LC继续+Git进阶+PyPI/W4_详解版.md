### 1. LeetCode 53：最大子数组和（Kadane算法）
```python
def max_subarray(nums):
    """
    Kadane算法：动态规划思想
    时间复杂度：O(n)，空间复杂度：O(1)
    
    思路：
    1. 用current_max记录当前子数组的最大和
    2. 用global_max记录全局最大和
    3. 遍历数组，对于每个元素：
       - current_max = max(nums[i], current_max + nums[i])
       - global_max = max(global_max, current_max)
    4. 返回global_max
    
    本质：如果当前子数组的和变成负数，就重新开始
    """
    if not nums:
        return 0
    
    current_max = nums[0]
    global_max = nums[0]
    
    for i in range(1, len(nums)):
        # 关键决策：要么从当前元素重新开始，要么继续累加
        current_max = max(nums[i], current_max + nums[i])
        # 更新全局最大值
        global_max = max(global_max, current_max)
    
    return global_max

# 示例运行过程：
# nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
# i=1: current_max = max(1, -2+1= -1) = 1, global_max = max(-2, 1) = 1
# i=2: current_max = max(-3, 1-3= -2) = -2, global_max = max(1, -2) = 1
# i=3: current_max = max(4, -2+4= 2) = 4, global_max = max(1, 4) = 4
# i=4: current_max = max(-1, 4-1= 3) = 3, global_max = max(4, 3) = 4
# i=5: current_max = max(2, 3+2= 5) = 5, global_max = max(4, 5) = 5
# i=6: current_max = max(1, 5+1= 6) = 6, global_max = max(5, 6) = 6
# i=7: current_max = max(-5, 6-5= 1) = 1, global_max = max(6, 1) = 6
# i=8: current_max = max(4, 1+4= 5) = 5, global_max = max(6, 5) = 6
# 返回：6
```

### 2. LeetCode 70：爬楼梯（动态规划）
```python
def climb_stairs(n):
    """
    动态规划解法：
    时间复杂度：O(n)，空间复杂度：O(1)
    
    思路：
    1. 定义dp[i]为爬到第i阶的方法数
    2. 初始状态：dp[0]=1, dp[1]=1 (爬到第0阶有1种方法：不爬)
    3. 状态转移：dp[i] = dp[i-1] + dp[i-2]
       - 可以从i-1阶爬1步上来
       - 可以从i-2阶爬2步上来
    4. 实际上就是斐波那契数列
    
    优化：只需要保存前两个状态
    """
    if n <= 1:
        return 1
    
    # 初始化前两个状态
    prev2 = 1  # dp[0]
    prev1 = 1  # dp[1]
    
    for i in range(2, n + 1):
        # 当前状态 = 前两个状态之和
        current = prev1 + prev2
        # 更新状态，为下一轮准备
        prev2, prev1 = prev1, current
    
    return prev1

# 更简洁的版本
def climb_stairs_clean(n):
    a, b = 1, 1
    for _ in range(n - 1):
        a, b = b, a + b
    return b

# 数学公式解法（了解即可）
def climb_stairs_math(n):
    """使用黄金比例公式（Binet's formula），不推荐实际使用"""
    import math
    sqrt5 = math.sqrt(5)
    phi = (1 + sqrt5) / 2
    return int((phi ** (n + 1) - (1 - phi) ** (n + 1)) / sqrt5)

# 递归解法（会超时，仅用于理解）
def climb_stairs_recursive(n):
    if n <= 1:
        return 1
    return climb_stairs_recursive(n - 1) + climb_stairs_recursive(n - 2)

# 带记忆化的递归解法
def climb_stairs_memo(n, memo=None):
    if memo is None:
        memo = {}
    if n in memo:
        return memo[n]
    if n <= 1:
        return 1
    memo[n] = climb_stairs_memo(n - 1, memo) + climb_stairs_memo(n - 2, memo)
    return memo[n]
```

### 3. LeetCode 121：买卖股票的最佳时机
```python
def max_profit(prices):
    """
    一次遍历解法：
    时间复杂度：O(n)，空间复杂度：O(1)
    
    思路：
    1. 初始化最小价格为第一天的价格
    2. 初始化最大利润为0
    3. 遍历价格：
       - 更新最小价格：min(min_price, current_price)
       - 计算当前利润：current_price - min_price
       - 更新最大利润：max(max_profit, current_profit)
    4. 返回最大利润
    
    本质：找到最低买入点，计算后续最高卖出点
    """
    if not prices or len(prices) < 2:
        return 0
    
    min_price = prices[0]
    max_profit_val = 0
    
    for price in prices[1:]:
        # 更新最低买入价格
        min_price = min(min_price, price)
        # 计算当前利润
        current_profit = price - min_price
        # 更新最大利润
        max_profit_val = max(max_profit_val, current_profit)
    
    return max_profit_val

# 暴力解法（对比理解）
def max_profit_brute_force(prices):
    """暴力枚举所有买入卖出组合，时间复杂度O(n²)"""
    n = len(prices)
    max_profit_val = 0
    
    for i in range(n):
        for j in range(i + 1, n):
            profit = prices[j] - prices[i]
            max_profit_val = max(max_profit_val, profit)
    
    return max_profit_val

# 示例运行过程：
# prices = [7, 1, 5, 3, 6, 4]
# day1: min_price=7, profit=0
# day2: min_price=1, profit=0 (1-7=-6)
# day3: min_price=1, profit=4 (5-1=4)
# day4: min_price=1, profit=4 (3-1=2)
# day5: min_price=1, profit=5 (6-1=5)
# day6: min_price=1, profit=5 (4-1=3)
# 返回：5
```

### 4. LeetCode 141：环形链表（快慢指针）
```python
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None

def has_cycle(head):
    """
    Floyd判圈算法（快慢指针）：
    时间复杂度：O(n)，空间复杂度：O(1)
    
    思路：
    1. 慢指针每次走一步，快指针每次走两步
    2. 如果链表有环，快慢指针最终会相遇
    3. 如果链表无环，快指针会先到达None
    
    证明：设环长为L，相遇时慢指针走了k步，快指针走了2k步
          2k - k = nL (n为整数)，所以k = nL
          即慢指针走了环长的整数倍，两者必然相遇
    """
    if not head or not head.next:
        return False
    
    slow = head
    fast = head.next
    
    while slow != fast:
        if not fast or not fast.next:
            return False
        slow = slow.next
        fast = fast.next.next
    
    return True

# 哈希表解法（对比理解）
def has_cycle_hash(head):
    """使用哈希表记录访问过的节点，时间复杂度O(n)，空间复杂度O(n)"""
    visited = set()
    current = head
    
    while current:
        if current in visited:
            return True
        visited.add(current)
        current = current.next
    
    return False

# 辅助函数：创建带环链表
def create_cycle_list(vals, pos):
    if not vals:
        return None
    nodes = [ListNode(val) for val in vals]
    for i in range(len(nodes) - 1):
        nodes[i].next = nodes[i + 1]
    if pos >= 0:
        nodes[-1].next = nodes[pos]
    return nodes[0]

# 示例运行过程：
# head = 3 → 2 → 0 → -4 → (指向2，形成环)
# slow: 3 → 2 → 0 → -4 → 2 → 0 → ...
# fast: 2 → -4 → 0 → 2 → -4 → 0 → ...
# 两者最终在环内相遇
```

### 2. Merge vs Rebase 区别

**Merge（合并）**：
```bash
# 创建新提交，保留分支历史
git checkout main
git merge feature/branch
```
- 优点：保留完整历史，不会重写提交
- 缺点：历史可能杂乱，包含很多合并提交
- 适用：公共分支（如main）、长期分支

**Rebase（变基）**：
```bash
# 将当前分支的提交"移到"目标分支的顶端
git checkout feature/branch
git rebase main
```
- 优点：历史线性清晰，易于理解
- 缺点：重写提交历史，可能引起混乱
- 适用：个人分支、尚未共享的分支

**黄金规则**：不要在公共分支上使用rebase！

### 3. 版本回退与恢复

**场景模拟操作**：
```bash
# 1. 初始化并创建commit1
git init
echo "print('Hello')" > app.py
git add app.py
git commit -m "Initial app"

# 2. 修改文件，创建commit2
echo "print('Hello World')" > app.py
git add app.py
git commit -m "Update greeting"

# 3. 再次修改文件（未提交）
echo "print('Hello, World!')" > app.py

# 4. 查看状态
git status
# 显示：app.py 已修改但未提交

# 5. 回退到commit1（硬重置）
git log --oneline  # 查看commit ID
git reset --hard HEAD~1  # 或使用commit ID

# 6. 使用git revert撤销commit2（如果已推送）
git revert HEAD  # 撤销最新提交
# 这会创建新的反向提交

# 7. 恢复工作区修改（如果在reset前未提交）
# 方法1：暂存并恢复
git stash  # 暂存修改
git stash pop  # 恢复暂存（在合适的分支）

# 方法2：直接恢复（如果知道文件内容）

# 8. 使用reflog找回"丢失"的提交
git reflog  # 查看所有操作历史
git reset --hard <commit-id>  # 恢复到某个状态
```

**命令对比**：
- `git reset --hard`：彻底丢弃，无法恢复（除非有reflog）
- `git reset --soft`：保留工作区和暂存区
- `git revert`：安全，创建新提交，适合已推送的提交
- `git stash`：临时保存未提交的修改

```python
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = left

def create_tree(arr):
    """从数组创建二叉树（层序遍历）"""
    if not arr:
        return None
    
    from collections import deque
    
    root = TreeNode(arr[0])
    queue = deque([root])
    i = 1
    
    while queue and i < len(arr):
        node = queue.popleft()
        
        # 左子节点
        if arr[i] is not None:
            node.left = TreeNode(arr[i])
            queue.append(node.left)
        i += 1
        
        # 右子节点
        if i < len(arr) and arr[i] is not None:
            node.right = TreeNode(arr[i])
            queue.append(node.right)
        i += 1
    
    return root

def preorder_traversal(root):
    """前序遍历"""
    result = []
    
    def traverse(node):
        if not node:
            return
        result.append(node.val)
        traverse(node.left)
        traverse(node.right)
    
    traverse(root)
    return result

    result = func(*args, **kwargs)
    end = time.perf_counter()
    
    elapsed = end - start
    
    print(f"函数 {func.__name__} 执行时间: {elapsed:.6f} 秒")
    print(f"输入: args={args}, kwargs={kwargs}")
    print(f"输出: {result}")
    
    return result, elapsed
```
