给你一个链表的头节点`head`，判断链表中是否有环。如果链表中存在环，则返回`true`；否则返回`false`。

        self.val = x
        self.next = None

    示例：
    输入：head = [3,2,0,-4], pos = 1（形成环）
    输出：true
    """

# 辅助函数：创建带环链表
def create_cycle_list(vals, pos):
    if not vals:
        return None
    nodes = [ListNode(val) for val in vals]
    for i in range(len(nodes)-1):
        nodes[i].next = nodes[i+1]
    if pos >= 0:
        nodes[-1].next = nodes[pos]
    return nodes[0]

# 测试
head1 = create_cycle_list([3,2,0,-4], 1)  # 有环

print(has_cycle(head1))  # True
print(has_cycle(head2))  # True
print(has_cycle(head3))  # False
